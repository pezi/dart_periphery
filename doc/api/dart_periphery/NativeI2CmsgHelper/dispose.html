void dispose ()

Frees all allocated memory blocks.

Implementation

void dispose() {
  if (_isFreed == true) {
    return;
  }
  _isFreed = true;
  var index = 0;
  for (var i = 0; i < size; ++i) {
    var msg = _messages.elementAt(index++);
    if (msg.ref.buf.address != 0) {
      free(msg.ref.buf);
    }
  }
  free(_messages);
}
